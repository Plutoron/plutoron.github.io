const fib = length => {
	const _wrap = new Array(length).fill('a')
    
    const isSecond = index => index === 0 || index === 1

	const fib = _wrap.reduce((pre, value, index) => {
		if(isSecond(index)) {
        	pre.push(1)
        } else {
        	pre.push(pre[index - 1] + pre[index - 2])
        }
      
        return pre
	}, [])
    
    return fib
}
