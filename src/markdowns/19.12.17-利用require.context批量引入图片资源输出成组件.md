### 利用require.context批量引入图片资源输出成组件

#### 目录结构
```
├── folder --------------------  一个目录
    ├── index.js  -------------  页面入口文件
    ├── icons  ----------------  图片资源目录
```

#### index.js
```
// how to 新增一个icon 
// 将 png 图片复制到 icons目录即可
// icon 文件名 命名 单一单词 或 驼峰写法 

// 定义 组件对象
const mods = {}
// 获取icons目录下所有png图片
const files = require.context('./icons', false, /\.png$/)
// 获取所有文件 相对路径 
// eg: ['./zip.png']
const reqPath = files.keys()

// 循环目录结构 组件赋值
reqPath.forEach(v => {
  // 获取 文件名 根据后续需求修改即可
  const name = v.replace('./', '').split('.')[0]
  // 组件名为 首字母大写的原文件名
  const modName = name.replace(name[0], name[0].toUpperCase())
  // 获取图片 
  // 前面的是直接取文件 后面的是引入组件的写法
  const img = files(v) || files(v).default 
  
  mods[modName] = ({ style }) => (
    <div style={{ width: 24, height: 24, ...(style || {}) }}>
      <img style={{ width: '100%', height: '100%' }} src={files(v)} />
    </div>
  )
})

console.log(mods)

module.exports = mods

```
