#### 整数反转

给出一个 32 位的有符号整数，你需要将这个整数中每位上的数字进行反转。

eg:
输入: 123
输出: 321

输入: -123
输出: -321

输入: 120
输出: 21

```javascript
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function(x) {
  if (typeof x === 'number') {
    if (x > Math.pow(2, 31) - 1 || x < -(Math.pow(2, 31))) return 0

    if (x >= 0) {
      return +(`${x}`.split('').reverse().join(''))
    } 

    if (x < 0) {
      return -(`${Math.sqrt(x * x)}`.split('').reverse().join(''))
    }
  } else {
    throw('x Type Error')
  }
};

console.log(reverse(-4294967296))
console.log(reverse(-210))
console.log(reverse(4294967296))
console.log(reverse(210))
```

```python
class Solution:
  def reverse(self, x: int) -> int:
    if x > pow(2, 31) - 1 or x < -(pow(2, 31)):
      return 0

    if x >= 0:
      _val = list(str(x))
      _val.reverse()
      reversed_val = ''.join(str(i) for i in _val)
      return int(reversed_val)

    if x < 0:
      _val = list(str(-x))
      _val.reverse()
      reversed_val = ''.join(str(i) for i in _val)
      return -int(reversed_val)

reverse = Solution().reverse(-120)

print(reverse)
```
