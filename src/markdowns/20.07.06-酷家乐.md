# prefill(n, m)

```
function prefill(n, m) {
  return new Array(n).fill(m)
}

function prefill(n, m) {
  if (n === 1) {
    return [m]
  } else {
    return [m, ...prefill(--n, m)]
  }
}

prefill(4, 3)
```
