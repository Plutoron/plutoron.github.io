		let index = 0
		let result = []
		let decimal = 0

			return _varr
		})

			const _sum = args.reduce((pre, value) => {
				return +(value[index] || 0) + pre
			}, decimal)

			if (index === maxLength - 1) {
				result.unshift(_sum)
				index ++
				break

		}
