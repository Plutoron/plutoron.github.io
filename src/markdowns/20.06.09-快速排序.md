#### 快速排序

```
var quickSort = (arr) => {  
  if (arr.length < 2) {
    return arr
  }

  var arr_back = [...arr]
  var cursor = 0
  var left = 0
  var right = arr.length - 1

  while(left < right) {
    // 左侧 大于 基数
    // 右侧 小于 基数 
    // 交换位置
    if (arr_back[left] <= arr_back[cursor]) {
      left += 1
    } 

    if (arr_back[right] > arr_back[cursor]) {
      right -= 1
    }

    if (arr_back[left] > arr_back[cursor] && arr_back[right] < arr_back[cursor]) {
      arr_back[left] = arr_back[right]
      arr_back[right] = arr[left]

      left += 1
      right -= 1
    }
  }

  if (arr_back[cursor] > arr_back[left]) {
    arr_back[cursor] = arr_back[left]
    arr_back[left] = arr[cursor]
  }

  var left_part = arr_back.slice(0, left)
  var right_part = arr_back.slice(left)

  return [...quickSort(left_part), ...quickSort(right_part)]
}

var arr = [1, 8, 6, 2, 7, 6, 5, 8, 4, 3]

console.log('result', quickSort(arr))
```
