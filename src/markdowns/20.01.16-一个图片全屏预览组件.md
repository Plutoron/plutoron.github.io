### 一个图片全屏预览组件

```
import { useState } from 'react'
import { Button } from 'antd'

const FullscreenPreviewImage = ({src}) => {
    const [visible, setVisible] = useState(false)

    return (
        [
            <Button shape="circle" icon="fullscreen" size="large" onClick={() => setVisible(true)} />,
            visible && <div 
                style={{
                    position: "fixed",
                    top: 0,
                    left: 0,
                    right: 0,
                    bottom: 0,
                    overflowY: 'auto',
                    zIndex: 1000,
                    backgroundColor: 'rgba(0,0,0, .6)',
                    
                }}
                className="box-align"
                onClick={() => setVisible(false)}
            >
                <img src={src} style={{cursor: 'zoom-out'}} />
            </div>
        ]
    )
}

export default FullscreenPreviewImage


// .box-align
.box-align {
    display: -webkit-box;
    -webkit-box-align: center;
    -webkit-box-pack: center;
    display: -moz-box;
    -moz-box-align: center;
    -moz-box-pack: center;
    display: -o-box;
    -o-box-align: center;
    -o-box-pack: center;
    display: -ms-box;
    -ms-box-align: center;
    -ms-box-pack: center;
    display: box;
    box-align: center;
    box-pack: center;
}
```
