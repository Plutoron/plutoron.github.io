#### 两数相加

给出两个 非空 的链表用来表示两个非负的整数。其中，它们各自的位数是按照 逆序 的方式存储的，并且它们的每个节点只能存储 一位 数字。

如果，我们将这两个数相加起来，则会返回一个新的链表来表示它们的和。

您可以假设除了数字 0 之外，这两个数都不会以 0 开头。

eg:
输入：(2 -> 4 -> 3) + (5 -> 6 -> 4)
输出：7 -> 0 -> 8
原因：342 + 465 = 807

```javascript
/**
 * Definition for singly-linked list.
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
function ListNode(val) {
  this.val = val;
  this.next = null;
}

var l1_2 = new ListNode(2)
var l1_4 = new ListNode(4)
var l1_3 = new ListNode(3)

l1_2.next = l1_4
l1_4.next = l1_3

var l2_5 = new ListNode(5)
var l2_6 = new ListNode(6)
var l2_4 = new ListNode(4)

l2_5.next = l2_6
l2_6.next = l2_4

var addTwoNumbers = function(l1, l2) {
  var result = new ListNode(0)

  var node = result

  var _l1 = l1
  var _l2 = l2

  while(_l1 && _l2) {
    var sum = node.val + _l1.val + _l2.val

    if (sum >= 10) {
      node.val = sum % 10
    } else {
      node.val = sum
    }

    _l1 = _l1.next
    _l2 = _l2.next

    if (!_l1) node.next = _l2
    if (!_l2) node.next = _l1

    if (_l1 && _l2) node.next = new ListNode(sum >= 10 ? 1 : 0)

    node = node.next
  }

  return result
};

addTwoNumbers(l1_2, l2_5)
```

class Solution:
  def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
    result = ListNode(0)
    node = result 
    _l1 = l1
    _l2 = l2

    while _l1 and _l2 :
      sum = node.val + _l1.val + _l2.val

      if sum >= 10:
        node.val = sum % 10
      else:
        node.val = sum

      _l1 = _l1.next
      _l2 = _l2.next

      if not _l1:
        node.next = l2

      if not _l2:
        node.next = l1

      if _l1 and _l2:
        node.next = ListNode(1 if sum >= 10 else  0)

      node = node.next
    
    return result

addTwoNumbers = Solution().addTwoNumbers(l1_2, l2_5)
