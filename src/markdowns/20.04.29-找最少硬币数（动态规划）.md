### 找最少硬币数（动态规划）

JS求组成n块的最少硬币数 (单个面值只有一块、五块、七块)

```
var getGroup = sum => {
  const result = {}
  let rest = 0

  // 整除
  if (sum % 7 === 0) {
    result['7'] = sum / 7
    return result
  }

  // 等于自身
  if (sum % 7 === sum) {
    result['7'] = 0
    rest = sum
  } else {
    result['7'] = (sum - sum % 7) / 7
    rest = sum % 7
  }
 
  if (rest % 5 === 0) {
    result['5'] = res / 5
    return result
  }

  if (rest % 5 === rest) {
    result['5'] = 0
    result['1'] = rest 
  } else {
    result['5'] = (rest - rest % 5) / 5
    result['1'] = rest % 5 
  }
    
  return result
}

getGroup(9)
```
