### shell 

```
#!/bin/bash
# 打 开发包
# eg sh ./deploy.sh
# 打 提测包
# eg: sh ./deploy.sh tag_name

# 打包命令
npm run build
# 复制 vendor 到 打包后的目录
cp -r ./vendors ./dist

# 获取当前时间戳
now=`date +%Y%m%d%H%M`

# 判断是否有tag_name
if [ x$1 != x ]
then
    #...有参数
    package_name="demo_v$1.tgz"
else
    #...没有参数
    package_name="demo_$now.tgz"
fi

tar zcvf $package_name ./dist
mkdir tgz
mv $package_name ./tgz
```
