### 处理千分位

```javascript
const formatNum = num => {
  const splitNum = () => {
    return (num + '').split('.')
  }

  const [left, right] = splitNum()
  const isGap = index => index % 3 === 0

  let leftStore = []
  let rightStore = []

  if (left) {
    let current = ''
    const MAX = left.length

    for (let i = 1; i <= MAX; i++) {
      current = left[MAX - i] + current

      if (i === MAX) {
        leftStore.unshift(current)
      }

      if (isGap(i)) {
        leftStore.unshift(current)
        current = ''
      } 
    }
  }

  if (right) {
    let current = ''
    const MAX = right.length

    for (let i = 1; i <= MAX; i++) {
      current = current + right[i - 1]

      if (i === MAX) {
        rightStore.push(current)
      }

      if (isGap(i)) {
        rightStore.push(current)
        current = ''
      } 
    }
  }

  const leftStr = leftStore.join(',')
  const rightStr = rightStore.join(',')

  return `${leftStr}${rightStr && '.' + rightStr}`
}

console.log(formatNum(1343354.3232))
console.log(formatNum(0.3232))
console.log(formatNum(1243242))
```
